Python Code (covid_dashboard.py):

from flask import Flask, render_template, jsonify
import requests
import json

# Create Flask app
app = Flask(__name__)

# COVID-19 API endpoint
COVID_API = "https://disease.sh/v3/covid-19/all"

@app.route('/')
def index():
  return render_template('index.html')

@app.route('/api/stats')
def get_stats():
  try:
    response = requests.get(COVID_API)
    data = response.json()
    return jsonify(data)
  except:
    return jsonify({"error": "Failed to fetch data"})

if __name__ == '__main__':
  app.run(debug=True, port=5000)
